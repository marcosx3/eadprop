ALTER SCHEMA sistemaead DEFAULT CHARACTER SET utf8mb4 DEFAULT COLLATE utf8mb4_unicode_ci;

DROP TABLE IF EXISTS TB_CURSOS;		/* TABELA DOS CURSOS */
DROP TABLE IF EXISTS TB_CATEGORIAS;	/* CATEGORIA DOS CURSOS */
DROP TABLE IF EXISTS TB_PAGAMENTOS;	/* PAGAMENTO */
DROP TABLE IF EXISTS TB_PESSOAS;	/* TABELA PARA USUARIOS */
DROP TABLE IF EXISTS TB_PROGRESSOS;	/* PARA ACOMPANHAR PROGRESSO DOS ESTUDANTES */

CREATE TABLE TB_CURSOS (
	COD_CURSO 	INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,	/* CHAVE PRIMÁRIA */
	COD_CATEGORIA	INT UNSIGNED NOT NULL DEFAULT 0,			/* CHAVE PRIMÁRIA */
	NOM_CURSO	VARCHAR(40),						/* NOME DO CURSO */
	DSC_CURSO	VARCHAR(120),						/* DESCRICAO DO CURSO */
	AUL_LN1_CURSO	VARCHAR(120),						/* AULA 1 DO CURSO */
	AUL_LN2_CURSO	VARCHAR(120),						/* AULA 2 DO CURSO */
	AUL_LN3_CURSO	VARCHAR(120),						/* AULA 3 DO CURSO */
	AUL_LN4_CURSO	VARCHAR(120),						/* AULA 4 DO CURSO */
	AUL_LN5_CURSO	VARCHAR(120),						/* AULA 5 DO CURSO */
	STS_CURSO	CHAR(1) DEFAULT 'A',					/* STATUS DO CURSO */
);

CREATE TABLE TB_CATEGORIAS (
	COD_CATEGORIA 	INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,	/* CHAVE PRIMÁRIA */
	NOM_CATEGORIA	VARCHAR(40),						/* NOME DA CATEGORIA */
	DSC_CATEGORIA	VARCHAR(120),						/* DESCRICAO DA CATEGORIA */
	STS_CURSO	CHAR(1) DEFAULT 'A',					/* STATUS DO CATEGORIA */
);
CREATE TABLE TB_PESSOAS (
	COD_PESSOA 	INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,	/* CHAVE PRIMÁRIA */
	NOM_PESSOA	VARCHAR(40),						/* NOME DA PESSOA*/
	CPF_PESSOA	VARCHAR(14),						/* CPF DA PESSOA*/
	CID_PESSOA	VARCHAR(40),						/* CIDADE DA PESSOA*/
	EST_PESSOA	VARCHAR(40),						/* ESTADO DA PESSOA*/
	DAT_NSC_PESSOA	DATE,							/* DATA DE NASCIMENTO DA PESSOA*/
	TIP_PESSOA	CHAR(1),						/* TIPO DE PESSOA*/
	STS_PESSOA	CHAR(1),						/* STATUS DA PESSOA*/
	
);
